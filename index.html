function buildGameCSVs(day, team) { // ------------------------- // GAME 1 (H1–H9) // ------------------------- const g1 = []; g1.push(`${day} Team ${team}`); g1.push("Game1"); for (let p = 1; p <= 4; p++) { const row = []; for (let h = 1; h <= 9; h++) { const input = document.querySelector( `[data-grid="g1"][data-player="${p}"][data-hole="${h}"]` ); row.push(input?.value || ""); } g1.push(row.join(",")); } const csvGame1 = g1.join("\n"); // ------------------------- // GAME 2 (H10–H18) // ------------------------- const g2 = []; g2.push(`${day} Team ${team}`); g2.push("Game2"); const labels = ["A", "B", "C", "D"]; for (let p = 1; p <= 4; p++) { const row = [labels[p - 1]]; for (let h = 10; h <= 18; h++) { const input = document.querySelector( `[data-grid="g2"][data-player="${p}"][data-hole="${h}"]` ); row.push(input?.value || ""); } g2.push(row.join(",")); } const csvGame2 = g2.join("\n"); // ------------------------- // GAME 3 (H1–H9) // ------------------------- const g3 = []; g3.push(`${day} Team ${team}`); g3.push("Game3"); for (let p = 1; p <= 4; p++) { const row = []; for (let h = 1; h <= 9; h++) { const input = document.querySelector( `[data-grid="g3"][data-player="${p}"][data-hole="${h}"]` ); row.push(input?.value || ""); } g3.push(row.join(",")); } const csvGame3 = g3.join("\n"); return { game1: csvGame1, game2: csvGame2, game3: csvGame3 }; }
